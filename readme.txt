# Comando para generar el cliente
java -jar C:\yisas\openapi-7.14.0\openapi-generator-cli.jar generate -i gpro-public-api-fixed.yml -g java -o gpro-client --library webclient --additional-properties=useJakartaEe=true,dateLibrary=java8,serializationLibrary=jackson,hideGenerationTimestamp=true --api-package org.jlobato.gpro.api --model-package org.jlobato.gpro.model --invoker-package org.jlobato.gpro.client --ignore-file-override=.openapi-generator-ignore --additional-properties=groupId=org.jlobato.gpro --additional-properties=artifactId=gpro-client --additional-properties=artifactVersion=1.0.0.RELEASE --additional-properties=artifactDescription="Java client por GPRO API generated by OpenAPI"

# Comando para generar el proxy
